CREATE TABLE CLIENTs
(	CLIENTSSN NUMBER(3,0) PRIMARY KEY, 
	CLIENTFNAME VARCHAR2(24 BYTE) NOT NULL,
	CLIENTLNAME VARCHAR2(24 BYTE) NOT NULL,
	CLIENTPHONENUMBER VARCHAR2(14 BYTE) NOT NULL,
	CLIENTGENDER CHAR(1 BYTE) NOT NULL, 
	CLIENTADDRESS VARCHAR2(50 BYTE) NOT NULL, 
	CLIENTNATIONALID NUMBER(14,0) NOT NULL
);
