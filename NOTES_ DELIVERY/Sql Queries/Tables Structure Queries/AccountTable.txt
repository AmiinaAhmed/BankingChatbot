CREATE TABLE ACCOUNT
( 	ACCOUNTID NUMBER(2,0)  PRIMARY KEY, 
	ACCOUNTBALANCE NUMBER(10,0) NOT NULL, 
	ACCOUNTPASSWORD VARCHAR2(24 BYTE) NOT NULL,
	ACCOUNTEMAILADDRESS VARCHAR2(40 BYTE) NOT NULL, 
	ACCOUNTTYPE VARCHAR2(25 BYTE) NOT NULL, 
	CLIENTSSN NUMBER(3,0) NOT NULL 
	REFERENCES CLIENT (CLIENTSSN) 
);
