CREATE TABLE BANKER
(	BANKERID NUMBER(2,0) PRIMARY KEY, 
	BANKERFNAME VARCHAR2(20 BYTE) NOT NULL,
	BANKERLNAME VARCHAR2(20 BYTE) NOT NULL ,
	BANKERNATIONALID NUMBER(14,0) NOT NULL,
	BANKERPHONENUMBER NUMBER(14,0) NOT NULL ,
	BANKERADDRESS VARCHAR2(50 BYTE) NOT NULL,
	BANKERGENDER CHAR(1 BYTE) NOT NULL
) ;
